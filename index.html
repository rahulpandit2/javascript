Variable declarations and selecting DOM elements

// Variables
let count = 0;       // can change
const siteName = "My Learning Repo"; // cannot change (constant)
var oldWay = "works but not preferred"; // function-scoped

// Data types (primitive)
let age = 21;                  // number
let name = "Dark Side";        // string
let isActive = true;           // boolean
let nothingHere = null;        // null
let notSet;                    // undefined

// Code block
if (isActive) {
  count = count + 1; // this line is inside a block

// DOM selection
const title  = document.getElementById("title");
const cards  = document.getElementsByClassName("card");
const firstP = document.querySelector("p");

  title.textContent = siteName;
